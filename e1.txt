    <title>Presidency University Registration</title> 
    <script> 
        function validateForm() { 
            let fname = document.getElementById("fname").value; 
            let lname = document.getElementById("lname").value; 
            let email = document.getElementById("email").value; 
            let password = document.getElementById("password").value; 
            let address = document.getElementById("address").value; 
            let phone = document.getElementById("phone").value; 
            let gender = document.querySelector('input[name="gender"]:checked'); 
            let namePattern = /^[A-Za-z]+$/; 
            let emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; 
            let passPattern = /^[A-Za-z0-9]+$/; 
            if (fname == "" || !namePattern.test(fname) || fname.length < 6) { 
                alert("First name must be at least 6 letters and only alphabets"); 
                return false; 
            } 
            if (!namePattern.test(lname)) { 
                alert("Last name must contain only alphabets"); 
                return false; 
            }  
            if (!emailPattern.test(email)) { 
                alert("Enter valid email (name@domain.com)"); 
                return false; 
            }
            if (password.length < 6 || !passPattern.test(password)) { 
                alert("Password must be at least 6 characters and only letters/numbers"); 
                return false; 
            } 
            if (address == "") { 
                alert("Address cannot be empty"); 
                return false; 
            } 
            if (!/^\d{10}$/.test(phone)) { 
                alert("Phone must be 10 digits only"); 
                return false; 
            } 
            if (!gender) { 
                alert("Select gender"); 
                return false; 
            } 
            alert("Registration done successfully"); 
            return true; 
        } 
    </script>
