from flask import Flask, request, jsonify

app = Flask(__name__)

def evaluate_expression(expression: str) -> int:
    """Evaluates a mathematical expression given as a string."""
    if not expression:
        return 0
    stack = []
    current_num = 0
    operator = '+'
    expression = expression.replace(" ", "") + '+'
    
    for char in expression:
        if char.isdigit():
            current_num = (current_num * 10) + int(char)
        elif char in "+-*/":
            if operator == '+': stack.append(current_num)
            elif operator == '-': stack.append(-current_num)
            elif operator == '*': stack.append(stack.pop() * current_num)
            elif operator == '/': stack.append(int(stack.pop() / current_num))
            operator = char
            current_num = 0
            
    return sum(stack)

@app.route('/')
def home():
    return "Send a POST request with a text file to the /upload endpoint!\n"

@app.route('/upload', methods=['POST'])
def calculate_from_file():
    if 'file' not in request.files:
        return jsonify({"error": "No file was sent in the request."}), 400
        
    file = request.files['file']
    
    if file.filename == '':
        return jsonify({"error": "File name is empty."}), 400
        
    if file:
        content = file.read().decode('utf-8')
        lines = content.splitlines()
        
        results = []
        for line in lines:
            line = line.strip()
            if line: 
                try:
                    res = evaluate_expression(line)
                    results.append({"equation": line, "result": res})
                except ZeroDivisionError:
                    results.append({"equation": line, "error": "Division by zero"})
                except Exception:
                    results.append({"equation": line, "error": "Invalid format"})
                    
        return jsonify({"batch_results": results})

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=8080)
